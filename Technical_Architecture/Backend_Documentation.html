MasalAI Backend Dokümantasyonu

📖 Swagger: api/docs
Auth: Bearer JWT
Runtime: Node.js + Express 4

İş Kuralları — Kod Akışı
BR-001
Masal Kelime Limiti — Maks. 1.300 kelime
Uygulama Katmanları
1. Prompt talimatı 2. Backend retry mantığı 3. DB CHECK constraint
// claude.service.ts — BR-001 retry mantığı const result = await generateStory(params); if (result.word_count > 1300) { // 1. deneme: kısaltma isteği gönder const retry = await retryWithShorter(params, result); if (retry.word_count > 1300) { // 2. deneme: 1300. kelimede kır retry.content = truncateAt(retry.content, 1300); retry.wasTruncated = true; } } // DB'de CHECK constraint: word_count <= 1300 // CONSTRAINT stories_word_count_check CHECK (word_count <= 1300)

// subscription.service.ts — BR-003 kota kontrolü async function checkAudioQuota(userId: string) { const sub = await prisma.subscription.findUnique({ where: { userId } }); if (sub.tier === 'free') throw 'AUDIO_NOT_AVAILABLE'; if (sub.tier === 'premium_yearly') { // Ay başı sıfırlama kontrolü if (isBefore(sub.monthResetAt, new Date())) { await resetMonthlyCount(userId); } if (sub.storiesThisMonth >= 20) throw 'AUDIO_LIMIT_REACHED'; } if (sub.tier === 'credits' && sub.credits < 1) throw 'INSUFFICIENT_CREDITS'; }

Auth Middleware — middleware/auth.ts
// Her korumalı route'a eklenir router.use(authMiddleware); function authMiddleware(req, res, next) { const token = req.headers.authorization?.split(' ')[1]; if (!token) return res.status(401).json({ error: 'UNAUTHORIZED' }); try { req.user = jwt.verify(token, process.env.JWT_SECRET); next(); } catch { res.status(401).json({ error: 'INVALID_TOKEN' }); } }
